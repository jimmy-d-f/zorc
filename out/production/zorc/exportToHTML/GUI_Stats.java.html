import java.io.Serializable;
import java.util.Random;

public class GUI_Stats implements Serializable {
    private int totalTime;
    private int totalWeight;
    private int totalCost;
    private int budget;
    private int weight;

    public GUI_Stats() {
        totalTime = 0;
        totalWeight = 0;
        totalCost = 0;
        budget = 0;
        weight = 0;
    }

    public void generateLimits() {
        weight = new Random().nextInt(300);
        budget = new Random().nextInt(1000);
    }

    public int getWeightLimit() { return weight; }
    public int getBudgetLimit() { return budget; }

    public void addRoomTime(int minutes) { totalTime += minutes; }

    public void addWeight(int kilograms) {
        totalWeight += kilograms;
    }

    public void addCost(int euros) {
        totalCost += euros;
    }

    public int getTotalTime() {
        return totalTime;
    }

    public int getTotalWeight() {
        return totalWeight;
    }

    public int getTotalCost() {
        return totalCost;
    }
}
